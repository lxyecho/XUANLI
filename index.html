        const policyData = {

            regions: [

            ]

        const map = new AMap.Map('map', {

            zoom: 4,

            center: [104.06, 35.67]

        });

 

        policyData.regions.forEach(region => {

            new AMap.Marker({

                position: region.coord,

                content: `<div style="background:#3498db;color:white;padding:6px;border-radius:4px">

                            ${region.name}<br>政策指数:${region.value}

                          </div>`

            }).setMap(map);

        });

 

        // 初始化Chart.js

        const ctx = document.getElementById('myChart').getContext('2d');

        new Chart(ctx, {

            type: 'bar',

            {

                labels: ['传统基建', '新基建', '城市更新'],

                datasets: [{

                    label: '投资占比变化 (%)',

                }]

            }

        });

 

        // 交互逻辑

        function showContent(type) {

            document.querySelectorAll('.nav-item').forEach(item => {

                item.classList.remove('active');

                item.querySelector('.animated-bar').style.width = '0';

            });

           

            event.currentTarget.classList.add('active');

            event.currentTarget.querySelector('.animated-bar').style.width = '100%';

        }
